1.backbutton增加退回后删除旧的调整拆分快照逻辑。这样可以确保每次点击应用时，创建的都是最新的调整拆分前的历史记录。
{
   "_obj": "select",
   "_target": [
      {
         "_ref": "snapshotClass",
         "_name": "调整拆分前"
      }
   ],
   "_isCommand": false
}
{
   "_obj": "delete",
   "_target": [
      {
         "_ref": "historyState",
         "_property": "currentHistoryState"
      }
   ],
   "_isCommand": false
}
